CREATE TABLE OBSERVACIONES (
    ID INTEGER NOT NULL,
    FECHAHORA TIME,
    COMANDA INTEGER,
    OBSERVACION VARCHAR(250),
    DESCIPCION VARCHAR(250));
alter table OBSERVACIONES
add constraint PK_OBSERVACIONES
primary key (ID);
CREATE GENERATOR GEN_OBSERVACIONES_ID;
SET TERM ^ ;

CREATE TRIGGER OBSERVACIONES_BI FOR OBSERVACIONES
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
  IF (NEW.ID IS NULL) THEN
    NEW.ID = GEN_ID(GEN_OBSERVACIONES_ID,1);
END
^

SET TERM ; ^
