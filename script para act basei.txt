CREATE TABLE USUARIOS (
    ID INTEGER NOT NULL,
    NOMBRE VARCHAR(70),
    CONTRASENIA VARCHAR(30),
    CLIENTES VARCHAR(10),
    OBRAS VARCHAR(10),
    FORMULAS VARCHAR(10),
    CAMIONES VARCHAR(10),
    OPERADORES VARCHAR(10),
    DOSIFICADORES VARCHAR(10),
    REIMPRESION VARCHAR(10),
    FOLIOREMISION VARCHAR(10));
CREATE GENERATOR GEN_USUARIOS_ID;
SET TERM ^ ;

CREATE TRIGGER USUARIOS_BI FOR USUARIOS
ACTIVE BEFORE INSERT POSITION 0
AS
BEGIN
  IF (NEW.ID IS NULL) THEN
    NEW.ID = GEN_ID(GEN_USUARIOS_ID,1);
END

^

SET TERM ; ^
